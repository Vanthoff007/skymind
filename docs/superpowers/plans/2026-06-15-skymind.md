# Foundation-Model-Guided UAV Navigation — Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a 3-layer ROS 2 system where a vision-language model replaces the classical frontier planner for indoor UAV exploration, evaluate it against geometric baselines in AirSim, and document results for an ETH Zurich master's application.

**Architecture:** Perception layer (LLaVA scene describer + numpy occupancy mapper) feeds natural language and grid state to a VLM planner (LLaMA-3.2-Vision via ollama), which outputs waypoints validated by a safety filter before being sent to PX4 via MAVROS. The entire stack is ROS 2 nodes with clean topic interfaces so each layer can be swapped or tested in isolation.

**Tech Stack:** Python 3.10, ROS 2 Humble, AirSim + PX4 SITL (Docker), MAVROS, ollama (LLaVA-1.6 + LLaMA-3.2-Vision), OpenCV, numpy, pytest, matplotlib/seaborn

```python
# ros2_ws/src/perception/perception/scene_describer.py
import base64
import cv2
import numpy as np
import ollama
import rclpy
from rclpy.node import Node
from sensor_msgs.msg import Image
from std_msgs.msg import String
from cv_bridge import CvBridge

PROMPT = (
    "Describe this UAV camera view for navigation. "
    "Focus on: open spaces, obstacles, doors, corridors, unexplored areas. "
    "Be concise. One to three sentences."
)
MODEL = "llava:7b"


def describe_scene(image: np.ndarray) -> str:
    """Convert numpy BGR image to base64 and query LLaVA. Returns description string."""
    _, buffer = cv2.imencode(".jpg", image)
    image_b64 = base64.b64encode(buffer).decode("utf-8")
    response = ollama.chat(
        model=MODEL,
        messages=[{"role": "user", "content": PROMPT, "images": [image_b64]}],
    )
    return response["message"]["content"]


class SceneDescriberNode(Node):
    def __init__(self):
        super().__init__("scene_describer")
        self._bridge = CvBridge()
        self._pub = self.create_publisher(String, "/perception/scene_description", 10)
        self.create_subscription(Image, "/camera/rgb", self._on_image, 1)
        self._last_description = ""
        # Throttle: process at most once every 0.5 s
        self._last_time = self.get_clock().now()
        self.get_logger().info("SceneDescriber ready")

    def _on_image(self, msg: Image):
        now = self.get_clock().now()
        if (now - self._last_time).nanoseconds < 500_000_000:
            return
        self._last_time = now
        img = self._bridge.imgmsg_to_cv2(msg, desired_encoding="bgr8")
        try:
            description = describe_scene(img)
        except Exception as e:
            self.get_logger().warn(f"LLaVA inference failed: {e}")
            return
        self._last_description = description
        out = String()
        out.data = description
        self._pub.publish(out)
        self.get_logger().info(f"Scene: {description[:80]}...")


def main(args=None):
    rclpy.init(args=args)
    node = SceneDescriberNode()
    rclpy.spin(node)
    node.destroy_node()
    rclpy.shutdown()
```

```python
# ros2_ws/src/perception/perception/occupancy_mapper.py
import math
import numpy as np
import rclpy
from rclpy.node import Node
from sensor_msgs.msg import Image
from geometry_msgs.msg import PoseStamped
from nav_msgs.msg import OccupancyGrid as RosOccupancyGrid
from cv_bridge import CvBridge


class OccupancyGrid:
    UNEXPLORED = 0
    EXPLORED = 1
    OBSTACLE = 2

    def __init__(self, size: int = 80, cell_size: float = 0.25):
        self.size = size
        self.cell_size = cell_size
        self.data = np.zeros((size, size), dtype=np.uint8)
        # World origin (0,0) maps to grid center
        self._offset = size // 2

    def world_to_grid(self, x: float, y: float):
        gx = int(x / self.cell_size) + self._offset
        gy = int(y / self.cell_size) + self._offset
        return np.clip(gx, 0, self.size - 1), np.clip(gy, 0, self.size - 1)

    def to_list(self):
        return self.data.flatten().tolist()


def mark_explored(grid: OccupancyGrid, drone_x: float, drone_y: float, radius: float = 1.5):
    """Mark all cells within radius metres of drone position as EXPLORED (unless OBSTACLE)."""
    cells = int(radius / grid.cell_size)
    cx, cy = grid.world_to_grid(drone_x, drone_y)
    for dx in range(-cells, cells + 1):
        for dy in range(-cells, cells + 1):
            if dx * dx + dy * dy <= cells * cells:
                gx = np.clip(cx + dx, 0, grid.size - 1)
                gy = np.clip(cy + dy, 0, grid.size - 1)
                if grid.data[gx][gy] != OccupancyGrid.OBSTACLE:
                    grid.data[gx][gy] = OccupancyGrid.EXPLORED


def update_from_depth(
    grid: OccupancyGrid,
    depth: np.ndarray,
    drone_x: float, drone_y: float, drone_z: float, yaw: float,
    obstacle_threshold: float = 0.5,
    cell_size: float = 0.25,
):
    """Project depth image pixels closer than obstacle_threshold to obstacle cells in the grid."""
    h, w = depth.shape
    fx = fy = w  # approximate focal length
    cx_px, cy_px = w / 2, h / 2
    # Sample every 10th pixel for speed
    ys, xs = np.mgrid[0:h:10, 0:w:10]
    zs = depth[ys, xs]
    mask = (zs > 0.01) & (zs < obstacle_threshold)
    if not mask.any():
        return
    xs_m = (xs[mask] - cx_px) * zs[mask] / fx
    ys_m = (ys[mask] - cy_px) * zs[mask] / fy
    # Rotate by yaw and add drone position
    cos_y, sin_y = math.cos(yaw), math.sin(yaw)
    world_x = drone_x + cos_y * zs[mask] - sin_y * xs_m
    world_y = drone_y + sin_y * zs[mask] + cos_y * xs_m
    for wx, wy in zip(world_x, world_y):
        gx, gy = grid.world_to_grid(wx, wy)
        grid.data[gx][gy] = OccupancyGrid.OBSTACLE


class OccupancyMapperNode(Node):
    def __init__(self):
        super().__init__("occupancy_mapper")
        self._bridge = CvBridge()
        self._grid = OccupancyGrid(size=80, cell_size=0.25)
        self._pose = None
        self._pub = self.create_publisher(RosOccupancyGrid, "/perception/occupancy_map", 1)
        self.create_subscription(Image, "/camera/depth", self._on_depth, 1)
        self.create_subscription(PoseStamped, "/mavros/local_position/pose", self._on_pose, 10)
        self.get_logger().info("OccupancyMapper ready")

    def _on_pose(self, msg: PoseStamped):
        self._pose = msg.pose

    def _on_depth(self, msg: Image):
        if self._pose is None:
            return
        depth = self._bridge.imgmsg_to_cv2(msg, desired_encoding="32FC1")
        x = self._pose.position.x
        y = self._pose.position.y
        z = self._pose.position.z
        # Extract yaw from quaternion
        q = self._pose.orientation
        yaw = math.atan2(2 * (q.w * q.z + q.x * q.y), 1 - 2 * (q.y ** 2 + q.z ** 2))
        update_from_depth(self._grid, depth, x, y, z, yaw)
        mark_explored(self._grid, x, y)
        self._publish_grid()

    def _publish_grid(self):
        msg = RosOccupancyGrid()
        msg.header.stamp = self.get_clock().now().to_msg()
        msg.header.frame_id = "map"
        msg.info.resolution = self._grid.cell_size
        msg.info.width = self._grid.size
        msg.info.height = self._grid.size
        msg.data = self._grid.to_list()
        self._pub.publish(msg)


def main(args=None):
    rclpy.init(args=args)
    node = OccupancyMapperNode()
    rclpy.spin(node)
    node.destroy_node()
    rclpy.shutdown()
```

```python
# ros2_ws/src/planning/planning/vlm_planner.py
import json
import re
import yaml
from pathlib import Path
from typing import Tuple
import numpy as np
import ollama
import rclpy
from rclpy.node import Node
from std_msgs.msg import String
from nav_msgs.msg import OccupancyGrid
from geometry_msgs.msg import PoseStamped

PROMPT_FILE = Path(__file__).parents[4] / "config" / "prompts" / "vlm_planner.yaml"
MODEL = "llama3.2-vision:11b"
MAX_RETRIES = 2


class WaypointParseError(Exception):
    pass


def parse_waypoint(raw: str) -> Tuple[float, float, float, float, str]:
    """Extract (x, y, z, yaw, reasoning) from model output. Handles JSON in prose."""
    match = re.search(r'\{[^{}]*"x"[^{}]*\}', raw, re.DOTALL)
    if not match:
        raise WaypointParseError(f"No JSON waypoint found in: {raw[:200]}")
    data = json.loads(match.group())
    return (
        float(data["x"]),
        float(data["y"]),
        float(data["z"]),
        float(data["yaw"]),
        str(data.get("reasoning", "")),
    )


def summarize_map(grid_data: np.ndarray) -> str:
    """Return a short string describing explored/obstacle coverage."""
    total = grid_data.size
    explored = int((grid_data == 1).sum())
    obstacles = int((grid_data == 2).sum())
    pct = explored / total * 100
    return f"{pct:.1f}% explored, {obstacles} obstacle cells."


def build_prompt(
    scene_description: str,
    map_summary: str,
    x: float, y: float, z: float, yaw: float,
    examples: list,
) -> str:
    parts = []
    for ex in examples:
        parts.append(
            f"Example:\nScene: {ex['scene']}\nMap: {ex['map_summary']}\n"
            f"Pose: {ex['pose']}\nOutput: {ex['response']}\n"
        )
    parts.append(
        f"Scene description: {scene_description}\n"
        f"Occupancy map summary: {map_summary}\n"
        f"Current drone pose: x={x:.2f} y={y:.2f} z={z:.2f} yaw={yaw:.2f}\n"
        f"Task: Explore as much of the indoor environment as possible.\n"
        f"Output the next waypoint as JSON with keys: x, y, z, yaw, reasoning."
    )
    return "\n".join(parts)


class VLMPlannerNode(Node):
    def __init__(self):
        super().__init__("vlm_planner")
        cfg = yaml.safe_load(PROMPT_FILE.read_text())
        self._examples = cfg.get("few_shot_examples", [])
        self._system_prompt = cfg.get("system", "")
        self._scene = ""
        self._grid = None
        self._pose = None
        self._waiting_for_completion = False

        self._pub_waypoint = self.create_publisher(PoseStamped, "/planner/waypoint", 1)
        self._pub_reasoning = self.create_publisher(String, "/planner/reasoning", 10)

        self.create_subscription(String, "/perception/scene_description", self._on_scene, 10)
        self.create_subscription(OccupancyGrid, "/perception/occupancy_map", self._on_map, 1)
        self.create_subscription(PoseStamped, "/mavros/local_position/pose", self._on_pose, 10)
        self.create_subscription(String, "/execution/waypoint_reached", self._on_reached, 10)

        self.get_logger().info("VLMPlanner ready")

    def _on_scene(self, msg: String):
        self._scene = msg.data

    def _on_map(self, msg: OccupancyGrid):
        self._grid = np.array(msg.data, dtype=np.uint8).reshape(msg.info.height, msg.info.width)

    def _on_pose(self, msg: PoseStamped):
        self._pose = msg.pose

    def _on_reached(self, msg: String):
        """Called when waypoint_controller signals waypoint completion. Triggers new plan."""
        self._waiting_for_completion = False
        if self._scene and self._grid is not None and self._pose is not None:
            self._plan_next_waypoint()

    def _plan_next_waypoint(self, rejection_context: str = ""):
        if self._waiting_for_completion:
            return
        x = self._pose.position.x
        y = self._pose.position.y
        z = self._pose.position.z
        q = self._pose.orientation
        import math
        yaw = math.atan2(2 * (q.w * q.z + q.x * q.y), 1 - 2 * (q.y ** 2 + q.z ** 2))
        map_summary = summarize_map(self._grid)
        scene = self._scene
        if rejection_context:
            scene += f" [Previous waypoint rejected: {rejection_context}]"
        prompt = build_prompt(scene, map_summary, x, y, z, yaw, self._examples)

        for attempt in range(MAX_RETRIES):
            try:
                response = ollama.chat(
                    model=MODEL,
                    messages=[
                        {"role": "system", "content": self._system_prompt},
                        {"role": "user", "content": prompt},
                    ],
                )
                raw = response["message"]["content"]
                wx, wy, wz, wyaw, reasoning = parse_waypoint(raw)
                self._publish_waypoint(wx, wy, wz, wyaw, reasoning)
                self._waiting_for_completion = True
                return
            except (WaypointParseError, Exception) as e:
                self.get_logger().warn(f"Attempt {attempt+1} failed: {e}")
        self.get_logger().error("All VLM planning attempts failed — hovering.")

    def _publish_waypoint(self, x, y, z, yaw, reasoning):
        import math
        msg = PoseStamped()
        msg.header.stamp = self.get_clock().now().to_msg()
        msg.header.frame_id = "map"
        msg.pose.position.x = x
        msg.pose.position.y = y
        msg.pose.position.z = z
        msg.pose.orientation.w = math.cos(yaw / 2)
        msg.pose.orientation.z = math.sin(yaw / 2)
        self._pub_waypoint.publish(msg)
        r = String()
        r.data = reasoning
        self._pub_reasoning.publish(r)
        self.get_logger().info(f"Waypoint: ({x:.2f},{y:.2f},{z:.2f}) — {reasoning[:60]}")


def main(args=None):
    rclpy.init(args=args)
    node = VLMPlannerNode()
    rclpy.spin(node)
    node.destroy_node()
    rclpy.shutdown()
```

```python
# ros2_ws/src/planning/planning/safety_filter.py
from enum import Enum
import numpy as np
import rclpy
from rclpy.node import Node
from geometry_msgs.msg import PoseStamped
from nav_msgs.msg import OccupancyGrid
from std_msgs.msg import String

MAX_RETRIES = 3


class SafetyResult(Enum):
    SAFE = "safe"
    OUT_OF_BOUNDS = "out_of_bounds"
    ALTITUDE_VIOLATION = "altitude_violation"
    COLLISION = "collision"


def is_waypoint_safe(
    x: float, y: float, z: float,
    grid: np.ndarray,
    bounds: dict,
    cell_size: float = 0.25,
) -> SafetyResult:
    if x < bounds["x_min"] or x > bounds["x_max"] or y < bounds["y_min"] or y > bounds["y_max"]:
        return SafetyResult.OUT_OF_BOUNDS
    if z < bounds["z_min"] or z > bounds["z_max"]:
        return SafetyResult.ALTITUDE_VIOLATION
    offset = grid.shape[0] // 2
    gx = int(x / cell_size) + offset
    gy = int(y / cell_size) + offset
    gx = np.clip(gx, 0, grid.shape[0] - 1)
    gy = np.clip(gy, 0, grid.shape[1] - 1)
    if grid[gx][gy] == 2:
        return SafetyResult.COLLISION
    return SafetyResult.SAFE


class SafetyFilterNode(Node):
    def __init__(self):
        super().__init__("safety_filter")
        self._grid = None
        self._retry_count = 0
        self._bounds = {
            "x_min": -10.0, "x_max": 10.0,
            "y_min": -10.0, "y_max": 10.0,
            "z_min": 0.5, "z_max": 3.0,
        }
        self._pub_safe = self.create_publisher(PoseStamped, "/execution/safe_waypoint", 1)
        self._pub_reject = self.create_publisher(String, "/safety_filter/rejection_reason", 10)
        self.create_subscription(OccupancyGrid, "/perception/occupancy_map", self._on_map, 1)
        self.create_subscription(PoseStamped, "/planner/waypoint", self._on_waypoint, 1)
        self.get_logger().info("SafetyFilter ready")

    def _on_map(self, msg: OccupancyGrid):
        self._grid = np.array(msg.data, dtype=np.uint8).reshape(msg.info.height, msg.info.width)

    def _on_waypoint(self, msg: PoseStamped):
        if self._grid is None:
            self.get_logger().warn("No occupancy map yet — rejecting waypoint")
            return
        x = msg.pose.position.x
        y = msg.pose.position.y
        z = msg.pose.position.z
        result = is_waypoint_safe(x, y, z, self._grid, self._bounds)
        if result == SafetyResult.SAFE:
            self._retry_count = 0
            self._pub_safe.publish(msg)
            self.get_logger().info(f"Waypoint ({x:.2f},{y:.2f},{z:.2f}) approved")
        else:
            self._retry_count += 1
            reason = result.value
            self.get_logger().warn(f"Waypoint rejected: {reason} (retry {self._retry_count}/{MAX_RETRIES})")
            r = String()
            r.data = reason
            self._pub_reject.publish(r)
            if self._retry_count >= MAX_RETRIES:
                self.get_logger().error("Max safety retries reached — hover in place")
                self._retry_count = 0


def main(args=None):
    rclpy.init(args=args)
    node = SafetyFilterNode()
    rclpy.spin(node)
    node.destroy_node()
    rclpy.shutdown()
```

```python
# ros2_ws/src/execution/execution/waypoint_controller.py
import math
import rclpy
from rclpy.node import Node
from geometry_msgs.msg import PoseStamped
from std_msgs.msg import String

POS_TOL = 0.2          # metres
YAW_TOL = math.radians(5)
PUBLISH_RATE_HZ = 20   # PX4 requires continuous setpoints in OFFBOARD mode


def is_waypoint_reached(
    curr_x, curr_y, curr_z, curr_yaw,
    goal_x, goal_y, goal_z, goal_yaw,
    pos_tol=POS_TOL, yaw_tol=YAW_TOL,
) -> bool:
    dist = math.sqrt((curr_x - goal_x)**2 + (curr_y - goal_y)**2 + (curr_z - goal_z)**2)
    yaw_err = abs(math.atan2(math.sin(goal_yaw - curr_yaw), math.cos(goal_yaw - curr_yaw)))
    return dist < pos_tol and yaw_err < yaw_tol


class WaypointControllerNode(Node):
    def __init__(self):
        super().__init__("waypoint_controller")
        self._goal = None
        self._curr_pose = None
        self._pub_setpoint = self.create_publisher(
            PoseStamped, "/mavros/setpoint_position/local", 10
        )
        self._pub_reached = self.create_publisher(String, "/execution/waypoint_reached", 10)
        self.create_subscription(PoseStamped, "/execution/safe_waypoint", self._on_goal, 1)
        self.create_subscription(
            PoseStamped, "/mavros/local_position/pose", self._on_pose, 10
        )
        self.create_timer(1.0 / PUBLISH_RATE_HZ, self._control_loop)
        self.get_logger().info("WaypointController ready")

    def _on_goal(self, msg: PoseStamped):
        self._goal = msg
        self.get_logger().info(
            f"New goal: ({msg.pose.position.x:.2f}, {msg.pose.position.y:.2f}, {msg.pose.position.z:.2f})"
        )

    def _on_pose(self, msg: PoseStamped):
        self._curr_pose = msg.pose

    def _control_loop(self):
        if self._goal is None:
            return
        # Always re-publish setpoint so PX4 stays in OFFBOARD mode
        self._pub_setpoint.publish(self._goal)
        if self._curr_pose is None:
            return
        q = self._curr_pose.orientation
        curr_yaw = math.atan2(2*(q.w*q.z + q.x*q.y), 1 - 2*(q.y**2 + q.z**2))
        gq = self._goal.pose.orientation
        goal_yaw = math.atan2(2*(gq.w*gq.z + gq.x*gq.y), 1 - 2*(gq.y**2 + gq.z**2))
        if is_waypoint_reached(
            self._curr_pose.position.x, self._curr_pose.position.y, self._curr_pose.position.z, curr_yaw,
            self._goal.pose.position.x, self._goal.pose.position.y, self._goal.pose.position.z, goal_yaw,
        ):
            self.get_logger().info("Waypoint reached")
            self._goal = None
            msg = String()
            msg.data = "reached"
            self._pub_reached.publish(msg)


def main(args=None):
    rclpy.init(args=args)
    node = WaypointControllerNode()
    rclpy.spin(node)
    node.destroy_node()
    rclpy.shutdown()
```

```python
# ros2_ws/src/planning/planning/frontier_planner.py
import math
from dataclasses import dataclass
import numpy as np
import rclpy
from rclpy.node import Node
from geometry_msgs.msg import PoseStamped
from nav_msgs.msg import OccupancyGrid
from std_msgs.msg import String

EXPLORED = 1
OBSTACLE = 2
UNEXPLORED = 0


@dataclass
class FrontierResult:
    found: bool
    x: float = 0.0
    y: float = 0.0
    z: float = 1.5


def find_nearest_frontier(
    grid: np.ndarray,
    drone_x: float,
    drone_y: float,
    cell_size: float = 0.25,
    cruise_z: float = 1.5,
) -> FrontierResult:
    """Find the nearest unexplored cell adjacent to an explored cell."""
    offset = grid.shape[0] // 2
    drone_gx = int(drone_x / cell_size) + offset
    drone_gy = int(drone_y / cell_size) + offset

    best_dist = float("inf")
    best_gx = best_gy = None

    rows, cols = grid.shape
    for gx in range(1, rows - 1):
        for gy in range(1, cols - 1):
            if grid[gx][gy] != UNEXPLORED:
                continue
            # Check if any 4-neighbour is EXPLORED
            neighbours = [grid[gx-1][gy], grid[gx+1][gy], grid[gx][gy-1], grid[gx][gy+1]]
            if EXPLORED in neighbours:
                dist = math.sqrt((gx - drone_gx)**2 + (gy - drone_gy)**2)
                if dist < best_dist:
                    best_dist = dist
                    best_gx, best_gy = gx, gy

    if best_gx is None:
        return FrontierResult(found=False)

    world_x = (best_gx - offset) * cell_size
    world_y = (best_gy - offset) * cell_size
    return FrontierResult(found=True, x=world_x, y=world_y, z=cruise_z)


class FrontierPlannerNode(Node):
    def __init__(self):
        super().__init__("frontier_planner")
        self._grid = None
        self._pose = None
        self._pub = self.create_publisher(PoseStamped, "/planner/waypoint", 1)
        self.create_subscription(OccupancyGrid, "/perception/occupancy_map", self._on_map, 1)
        self.create_subscription(PoseStamped, "/mavros/local_position/pose", self._on_pose, 10)
        self.create_subscription(String, "/execution/waypoint_reached", self._on_reached, 10)
        self.get_logger().info("FrontierPlanner ready")

    def _on_map(self, msg: OccupancyGrid):
        self._grid = np.array(msg.data, dtype=np.uint8).reshape(msg.info.height, msg.info.width)

    def _on_pose(self, msg: PoseStamped):
        self._pose = msg.pose

    def _on_reached(self, _msg):
        if self._grid is None or self._pose is None:
            return
        result = find_nearest_frontier(
            self._grid, self._pose.position.x, self._pose.position.y
        )
        if not result.found:
            self.get_logger().info("No frontier found — exploration complete")
            return
        goal = PoseStamped()
        goal.header.stamp = self.get_clock().now().to_msg()
        goal.header.frame_id = "map"
        goal.pose.position.x = result.x
        goal.pose.position.y = result.y
        goal.pose.position.z = result.z
        goal.pose.orientation.w = 1.0
        self._pub.publish(goal)
        self.get_logger().info(f"Frontier: ({result.x:.2f}, {result.y:.2f})")


def main(args=None):
    rclpy.init(args=args)
    node = FrontierPlannerNode()
    rclpy.spin(node)
    node.destroy_node()
    rclpy.shutdown()
```

```python
# ros2_ws/src/evaluation/evaluation/metrics_logger.py
import csv
import time
from dataclasses import dataclass, field
from pathlib import Path
from typing import List, Optional
import numpy as np
import rclpy
from rclpy.node import Node
from nav_msgs.msg import OccupancyGrid
from std_msgs.msg import String

RESULTS_DIR = Path(__file__).parents[4] / "evaluation" / "results"
EXPLORED = 1


def compute_coverage(grid: np.ndarray) -> float:
    """Returns fraction of non-obstacle cells that are explored."""
    non_obstacle = (grid != 2).sum()
    if non_obstacle == 0:
        return 0.0
    explored = (grid == EXPLORED).sum()
    return float(explored) / float(non_obstacle)


@dataclass
class TrialMetrics:
    collision_count: int = 0
    planner_failures: int = 0
    coverage_samples: List[tuple] = field(default_factory=list)  # (t, coverage)

    def record_collision(self):
        self.collision_count += 1

    def record_planner_failure(self):
        self.planner_failures += 1

    def record_coverage_sample(self, t: float, coverage: float):
        self.coverage_samples.append((t, coverage))

    def time_to_coverage(self, threshold: float) -> Optional[float]:
        for t, cov in self.coverage_samples:
            if cov >= threshold:
                return t
        return None

    def final_coverage(self) -> float:
        if not self.coverage_samples:
            return 0.0
        return self.coverage_samples[-1][1]

    def to_dict(self, trial_id: int, condition: str) -> dict:
        return {
            "trial_id": trial_id,
            "condition": condition,
            "final_coverage": self.final_coverage(),
            "time_to_50pct": self.time_to_coverage(0.5),
            "time_to_75pct": self.time_to_coverage(0.75),
            "time_to_90pct": self.time_to_coverage(0.90),
            "collision_count": self.collision_count,
            "planner_failures": self.planner_failures,
        }


class MetricsLoggerNode(Node):
    def __init__(self):
        super().__init__("metrics_logger")
        self.declare_parameter("trial_id", 0)
        self.declare_parameter("condition", "vlm")
        self._trial_id = self.get_parameter("trial_id").value
        self._condition = self.get_parameter("condition").value
        self._metrics = TrialMetrics()
        self._grid = None
        self._start_time = time.time()
        RESULTS_DIR.mkdir(parents=True, exist_ok=True)

        self.create_subscription(OccupancyGrid, "/perception/occupancy_map", self._on_map, 1)
        self.create_subscription(String, "/safety_filter/rejection_reason", self._on_rejection, 10)
        self.create_timer(5.0, self._sample_coverage)
        self.create_subscription(String, "/evaluation/end_trial", self._on_end, 10)
        self.get_logger().info(f"MetricsLogger ready — trial {self._trial_id}, condition {self._condition}")

    def _on_map(self, msg: OccupancyGrid):
        self._grid = np.array(msg.data, dtype=np.uint8).reshape(msg.info.height, msg.info.width)

    def _on_rejection(self, msg: String):
        if "max" in msg.data.lower() or "retries" in msg.data.lower():
            self._metrics.record_planner_failure()

    def _sample_coverage(self):
        if self._grid is None:
            return
        t = time.time() - self._start_time
        cov = compute_coverage(self._grid)
        self._metrics.record_coverage_sample(t, cov)
        self.get_logger().info(f"t={t:.0f}s coverage={cov*100:.1f}%")

    def _on_end(self, _msg):
        row = self._metrics.to_dict(self._trial_id, self._condition)
        out_path = RESULTS_DIR / f"trial_{self._trial_id:03d}_{self._condition}.csv"
        with open(out_path, "w", newline="") as f:
            writer = csv.DictWriter(f, fieldnames=row.keys())
            writer.writeheader()
            writer.writerow(row)
        self.get_logger().info(f"Trial saved to {out_path}")


def main(args=None):
    rclpy.init(args=args)
    node = MetricsLoggerNode()
    rclpy.spin(node)
    node.destroy_node()
    rclpy.shutdown()
```

```python
# evaluation/analyse.py
"""
Load all trial CSVs from evaluation/results/, compute summary stats,
run t-tests, and save plots to evaluation/plots/.

Usage:
    python3 evaluation/analyse.py
"""
import csv
import glob
from pathlib import Path
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from scipy import stats

RESULTS_DIR = Path(__file__).parent / "results"
PLOTS_DIR = Path(__file__).parent / "plots"
PLOTS_DIR.mkdir(exist_ok=True)

CONDITIONS = ["vlm", "frontier"]
METRICS = ["final_coverage", "time_to_50pct", "time_to_75pct", "collision_count"]


def load_results() -> dict:
    data = {c: [] for c in CONDITIONS}
    for path in glob.glob(str(RESULTS_DIR / "*.csv")):
        with open(path) as f:
            rows = list(csv.DictReader(f))
            for row in rows:
                cond = row["condition"]
                if cond in data:
                    data[cond].append({k: float(v) if v != "" else None for k, v in row.items()})
    return data


def print_summary(data: dict):
    print("\n=== Summary ===")
    for cond, rows in data.items():
        if not rows:
            continue
        coverages = [r["final_coverage"] for r in rows if r["final_coverage"] is not None]
        collisions = [r["collision_count"] for r in rows if r["collision_count"] is not None]
        print(f"\n{cond.upper()} (n={len(rows)})")
        print(f"  Coverage: mean={np.mean(coverages)*100:.1f}% std={np.std(coverages)*100:.1f}%")
        print(f"  Collisions: mean={np.mean(collisions):.2f}")


def run_ttest(data: dict):
    if "vlm" not in data or "frontier" not in data:
        return
    vlm_cov = [r["final_coverage"] for r in data["vlm"] if r["final_coverage"]]
    fr_cov = [r["final_coverage"] for r in data["frontier"] if r["final_coverage"]]
    if len(vlm_cov) < 2 or len(fr_cov) < 2:
        print("Not enough data for t-test yet.")
        return
    t, p = stats.ttest_ind(vlm_cov, fr_cov)
    print(f"\nt-test (coverage): t={t:.3f}, p={p:.4f}")
    if p < 0.05:
        print("  → Statistically significant difference (p < 0.05)")
    else:
        print("  → No significant difference yet")


def plot_coverage_boxplot(data: dict):
    fig, ax = plt.subplots(figsize=(8, 5))
    plot_data = []
    labels = []
    for cond in CONDITIONS:
        vals = [r["final_coverage"] * 100 for r in data.get(cond, []) if r["final_coverage"]]
        if vals:
            plot_data.append(vals)
            labels.append(cond.upper())
    if not plot_data:
        print("No data to plot yet.")
        return
    ax.boxplot(plot_data, labels=labels)
    ax.set_ylabel("Final Coverage (%)")
    ax.set_title("Exploration Coverage: VLM Planner vs Frontier Baseline")
    ax.grid(True, alpha=0.3)
    out = PLOTS_DIR / "coverage_boxplot.png"
    fig.savefig(out, dpi=150, bbox_inches="tight")
    print(f"Saved {out}")


if __name__ == "__main__":
    data = load_results()
    print_summary(data)
    run_ttest(data)
    plot_coverage_boxplot(data)
```

```bash
  sleep 10
done
```
